function showDashboard() {
  document.getElementById("login").classList.remove("active");
  document.getElementById("dashboard").classList.add("active");
}

function logout() {
  document.getElementById("dashboard").classList.remove("active");
  document.getElementById("login").classList.add("active");
}

function update(device) {
  const value = event.target.value;
  document.getElementById(`${device}-${device === 'ac' ? 'temp' : device === 'light' ? 'bright' : 'speed'}`).textContent = value;
}

function toggle(btn) {
  const isOn = btn.textContent.includes("OFF");
  btn.textContent = isOn ? "Turn ON" : "Turn OFF";
  btn.style.background = isOn ? "#444" : "orange";
}
